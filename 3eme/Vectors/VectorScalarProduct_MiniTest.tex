For $\overrightarrow{a} = \left(\begin{matrix} a_1 \\ a_2 \end{matrix} \right)$ and $\overrightarrow{b} = \left(\begin{matrix} b_1 \\ b_2 \end{matrix} \right)$ the \textbf{scalar product} is $ a \bullet b = \| \overrightarrow{a} \| \cdot \| \overrightarrow{b} \| \cdot \cos \theta = a_1 \cdot b_1 + a_2 \cdot b_2$.

\begin{questions} 
\question Take $\overrightarrow{x} = \left(\begin{matrix} 3 \\ 4 \end{matrix} \right)$ and $\overrightarrow{y} = \left(\begin{matrix} 12 \\ -5 \end{matrix} \right)$.
\begin{parts}
\part[1] Calculate the scalar product of $\overrightarrow{x}$ and $\overrightarrow{y}$.
\fillwithdottedlines{1cm} 
\part[4] Calculate the angle from $\overrightarrow{x}$ to $\overrightarrow{y}$.
\fillwithdottedlines{5cm} 
\end{parts}

\question[4] Given that $\left(\begin{matrix} \sqrt{3} \\ -6 \end{matrix} \right) \perp \left(\begin{matrix} \sqrt{27} \\ \lambda \end{matrix} \right)$, find $\lambda$.
\fillwithdottedlines{3cm}

\question[3] Simplify $\overrightarrow{a} \bullet (\overrightarrow{b} + \overrightarrow{c}) -  \overrightarrow{b} \bullet (\overrightarrow{a} -\overrightarrow{c})$, given that $\overrightarrow{b} \perp \overrightarrow{c}$.
\hfill \emph{Do not use column notation.}
\fillwithdottedlines{4cm}
